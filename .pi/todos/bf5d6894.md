## Tasks
1. Update discount code validation to check referral codes
2. Pass referral info through Stripe metadata
3. Update webhook to credit referrer

## Checkout Integration
```typescript
// app/api/discount/route.ts
import { validateReferralCode } from "@/lib/referral";

export async function POST(request: Request) {
  const { code, appId } = await request.json();
  const user = await getCurrentUser(env);
  
  // First check discount_codes table (existing logic)
  let discount = await queries.getDiscountCode(code, env);
  
  // If not found, check referral codes
  if (!discount && user) {
    const referral = await validateReferralCode(code, user.id, env);
    if (referral) {
      discount = {
        type: "referral",
        percent_off: referral.discount_percent,
        referral_code_id: referral.id,
        referrer_user_id: referral.user_id,
        referrer_reward_cents: referral.reward_cents,
      };
    }
  }
  
  return Response.json({ discount });
}
```

## Webhook Handler
```typescript
// On successful checkout with referral
if (metadata.referral_code_id) {
  // Record the referral
  await execute(`
    INSERT INTO referrals (id, code_id, referrer_user_id, referred_user_id, purchase_id, discount_applied_cents, reward_credited_cents)
    VALUES (?, ?, ?, ?, ?, ?, ?)
  `, [nanoid(), metadata.referral_code_id, metadata.referrer_user_id, userId, purchaseId, discountAmount, metadata.referrer_reward_cents]);
  
  // Credit the referrer
  await execute(`
    UPDATE user SET referral_credit_cents = referral_credit_cents + ? WHERE id = ?
  `, [metadata.referrer_reward_cents, metadata.referrer_user_id]);
  
  // Increment times_used
  await execute(`
    UPDATE referral_codes SET times_used = times_used + 1 WHERE id = ?
  `, [metadata.referral_code_id]);
}
```
