## Tasks
1. Create `app/bundle/page.tsx`
2. Fetch bundle info and included apps
3. Calculate savings vs individual purchase
4. Show app icons grid
5. Add purchase CTA

## Implementation
```typescript
// app/bundle/page.tsx
import { getEnv } from "@/lib/cloudflare-context";
import { getCurrentUser } from "@/lib/auth/middleware";
import { SiteNav } from "@/components/site-nav";
import { BundlePurchaseCard } from "./bundle-purchase-card";

async function getBundle() {
  const env = getEnv();
  
  // Get the main bundle
  const bundle = await env.DB.prepare(`
    SELECT * FROM bundles WHERE slug = 'all-apps' AND is_active = 1
  `).first();
  
  if (!bundle) return null;
  
  // Get all apps (for "includes future" bundles)
  const apps = await env.DB.prepare(`
    SELECT id, slug, name, icon_url, min_price_cents
    FROM apps WHERE is_published = 1
  `).all();
  
  // Calculate individual total
  const individualTotal = apps.results.reduce(
    (sum, app) => sum + (app.min_price_cents || 0), 
    0
  );
  
  return {
    ...bundle,
    apps: apps.results,
    individualTotal,
    savings: individualTotal - bundle.price_cents,
  };
}

export default async function BundlePage() {
  const env = getEnv();
  const [user, bundle] = await Promise.all([
    env ? getCurrentUser(env) : null,
    getBundle(),
  ]);
  
  if (!bundle) {
    return <div>No bundle available</div>;
  }
  
  return (
    <>
      <SiteNav user={user} />
      
      <main className="bundle-page">
        <header className="bundle-header">
          <h1 className="bundle-header__title">{bundle.name}</h1>
          <p className="bundle-header__subtitle">
            Everything we make. Forever.
          </p>
        </header>
        
        <div className="bundle-apps">
          {bundle.apps.map(app => (
            <div key={app.id} className="bundle-app">
              <img src={app.icon_url} alt={app.name} />
              <span>{app.name}</span>
            </div>
          ))}
          <div className="bundle-app bundle-app--future">
            <span>+</span>
            <span>Future Apps</span>
          </div>
        </div>
        
        <div className="bundle-value">
          <p>{bundle.apps.length} apps included</p>
          <p>Worth ${(bundle.individualTotal / 100).toFixed(2)} individually</p>
          <p className="bundle-value__savings">
            Save ${(bundle.savings / 100).toFixed(2)}
          </p>
        </div>
        
        <BundlePurchaseCard 
          bundleId={bundle.id}
          priceCents={bundle.price_cents}
          isAuthenticated={!!user}
        />
        
        <ul className="bundle-benefits">
          <li>✓ One-time purchase</li>
          <li>✓ Includes all future releases</li>
          <li>✓ All platforms (macOS + iOS)</li>
          <li>✓ Lifetime updates</li>
        </ul>
      </main>
    </>
  );
}
```
