```typescript
// app/dashboard/page.tsx

// Check if user owns a bundle
const bundleOwnership = await env.DB.prepare(`
  SELECT b.name, b.includes_future_apps, p.created_at as purchased_at
  FROM purchases p
  JOIN bundles b ON p.bundle_id = b.id
  WHERE p.user_id = ? AND p.status = 'completed'
  LIMIT 1
`).bind(user.id).first();

// In the render
{bundleOwnership && (
  <div className="bundle-ownership-card">
    <h3>🎁 {bundleOwnership.name}</h3>
    <p>You own the complete bundle!</p>
    {bundleOwnership.includes_future_apps && (
      <p className="bundle-note">
        New apps will be automatically added to your library.
      </p>
    )}
  </div>
)}
```
