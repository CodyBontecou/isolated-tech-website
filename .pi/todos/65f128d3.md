## Tasks
1. Check user credit during checkout
2. Apply credit as discount
3. Deduct used credit after successful purchase
4. Handle partial credit usage

## Checkout Logic
```typescript
// In checkout API route

// Get user's credit balance
const user = await env.DB.prepare(
  "SELECT referral_credit_cents FROM user WHERE id = ?"
).bind(userId).first();

const availableCredit = user?.referral_credit_cents || 0;

// Calculate final price
let finalPrice = appPriceCents;

// Apply discount code first (if any)
if (discountPercent) {
  finalPrice = Math.round(finalPrice * (1 - discountPercent / 100));
}

// Apply credit (up to remaining price)
const creditToApply = Math.min(availableCredit, finalPrice);
finalPrice -= creditToApply;

// If price is now 0, skip Stripe
if (finalPrice === 0) {
  // Create purchase directly
  await createPurchase({ userId, appId, amountCents: 0, status: "completed" });
  
  // Deduct credit
  await execute(
    "UPDATE user SET referral_credit_cents = referral_credit_cents - ? WHERE id = ?",
    [creditToApply, userId]
  );
  
  return Response.json({ success: true, redirect: "/dashboard" });
}

// Otherwise create Stripe session with remaining amount
const session = await stripe.checkout.sessions.create({
  // ...
  line_items: [{
    price_data: {
      unit_amount: finalPrice,
      // ...
    },
  }],
  metadata: {
    credit_applied_cents: creditToApply.toString(),
    // ...
  },
});
```

## Webhook Handler
```typescript
// After successful purchase
const creditApplied = parseInt(metadata.credit_applied_cents || "0");
if (creditApplied > 0) {
  await execute(
    "UPDATE user SET referral_credit_cents = referral_credit_cents - ? WHERE id = ?",
    [creditApplied, userId]
  );
}
```
