## Tasks
1. Create migration for referral_codes table
2. Create referrals tracking table
3. Add referral_credit_cents to user table
4. Apply migrations

## Migration SQL
```sql
-- migrations/XXXX_add_referrals.sql

-- Referral codes owned by users
CREATE TABLE IF NOT EXISTS referral_codes (
  id TEXT PRIMARY KEY,
  user_id TEXT NOT NULL,
  code TEXT UNIQUE NOT NULL,
  discount_percent INTEGER DEFAULT 20,
  discount_cents INTEGER,
  reward_cents INTEGER DEFAULT 500,  -- $5 reward per referral
  times_used INTEGER DEFAULT 0,
  max_uses INTEGER,  -- NULL = unlimited
  is_active INTEGER DEFAULT 1,
  created_at TEXT DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES user(id) ON DELETE CASCADE
);

-- Track successful referrals
CREATE TABLE IF NOT EXISTS referrals (
  id TEXT PRIMARY KEY,
  code_id TEXT NOT NULL,
  referrer_user_id TEXT NOT NULL,
  referred_user_id TEXT NOT NULL,
  purchase_id TEXT NOT NULL,
  discount_applied_cents INTEGER NOT NULL,
  reward_credited_cents INTEGER NOT NULL,
  created_at TEXT DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (code_id) REFERENCES referral_codes(id),
  FOREIGN KEY (referrer_user_id) REFERENCES user(id),
  FOREIGN KEY (referred_user_id) REFERENCES user(id),
  FOREIGN KEY (purchase_id) REFERENCES purchases(id)
);

-- Add credit tracking to users
ALTER TABLE user ADD COLUMN referral_credit_cents INTEGER DEFAULT 0;

-- Indexes
CREATE INDEX idx_referral_codes_user ON referral_codes(user_id);
CREATE INDEX idx_referral_codes_code ON referral_codes(code);
CREATE INDEX idx_referrals_referrer ON referrals(referrer_user_id);
```
