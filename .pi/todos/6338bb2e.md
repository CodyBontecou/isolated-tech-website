## Code Changes
```typescript
// app/error.tsx
import * as Sentry from "@sentry/cloudflare";

useEffect(() => {
  Sentry.captureException(error, {
    tags: { digest: error.digest },
    extra: { componentStack: error.componentStack },
  });
}, [error]);
```
