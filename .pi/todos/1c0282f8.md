## Tasks
1. Create `app/api/checkout/bundle/route.ts`
2. Validate user authentication
3. Check for existing bundle ownership
4. Create Stripe checkout session
5. Include bundle metadata for webhook

## Implementation
```typescript
// app/api/checkout/bundle/route.ts
import { getEnv } from "@/lib/cloudflare-context";
import { getCurrentUser } from "@/lib/auth/middleware";
import { getStripe } from "@/lib/stripe";

export async function POST(request: Request) {
  const env = getEnv();
  const user = await getCurrentUser(env);
  
  if (!user) {
    return Response.json({ error: "Authentication required" }, { status: 401 });
  }
  
  const { bundleId } = await request.json();
  
  // Get bundle
  const bundle = await env.DB.prepare(
    "SELECT * FROM bundles WHERE id = ? AND is_active = 1"
  ).bind(bundleId).first();
  
  if (!bundle) {
    return Response.json({ error: "Bundle not found" }, { status: 404 });
  }
  
  // Check if user already owns the bundle
  const existingBundle = await env.DB.prepare(
    "SELECT id FROM purchases WHERE user_id = ? AND bundle_id = ? AND status = 'completed'"
  ).bind(user.id, bundleId).first();
  
  if (existingBundle) {
    return Response.json({ error: "You already own this bundle" }, { status: 400 });
  }
  
  // Get all apps to include in metadata
  const apps = await env.DB.prepare(
    "SELECT id FROM apps WHERE is_published = 1"
  ).all();
  
  const appIds = apps.results.map(a => a.id).join(",");
  
  // Create Stripe session
  const stripe = getStripe(env);
  const session = await stripe.checkout.sessions.create({
    mode: "payment",
    customer_email: user.email,
    line_items: [{
      price_data: {
        currency: "usd",
        unit_amount: bundle.price_cents,
        product_data: {
          name: bundle.name,
          description: bundle.description || "All apps bundle",
        },
      },
      quantity: 1,
    }],
    success_url: `${env.SITE_URL}/dashboard?purchased=bundle`,
    cancel_url: `${env.SITE_URL}/bundle`,
    metadata: {
      type: "bundle",
      bundle_id: bundleId,
      user_id: user.id,
      app_ids: appIds,
      includes_future: bundle.includes_future_apps ? "true" : "false",
    },
  });
  
  return Response.json({ url: session.url });
}
```
