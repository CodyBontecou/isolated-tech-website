## Dependencies

```bash
npm install satori @resvg/resvg-wasm
```

## Font Setup
```typescript
// lib/og-fonts.ts
export async function loadFonts() {
  const spaceMono = await fetch(
    "https://fonts.gstatic.com/s/spacemono/v12/i7dPIFZifjKcF5UAWdDRYEF8RQ.woff"
  ).then(res => res.arrayBuffer());
  
  return [
    {
      name: "Space Mono",
      data: spaceMono,
      weight: 400,
      style: "normal",
    },
  ];
}
```

## OG Component
```tsx
// lib/og-image.tsx
export function OGImage({ 
  name, 
  tagline, 
  iconUrl 
}: { 
  name: string; 
  tagline?: string; 
  iconUrl?: string;
}) {
  return (
    <div style={{
      width: 1200,
      height: 630,
      background: "#0a0a0a",
      display: "flex",
      alignItems: "center",
      padding: 80,
    }}>
      {iconUrl && (
        <img src={iconUrl} width={200} height={200} style={{ borderRadius: 40 }} />
      )}
      <div style={{ marginLeft: 40, flex: 1 }}>
        <div style={{ color: "#fff", fontSize: 64, fontFamily: "Space Mono" }}>
          {name}
        </div>
        {tagline && (
          <div style={{ color: "#666", fontSize: 32, marginTop: 16 }}>
            {tagline}
          </div>
        )}
      </div>
      <div style={{ position: "absolute", bottom: 40, left: 80, color: "#444", fontSize: 24 }}>
        ISOLATED.TECH
      </div>
    </div>
  );
}
```
