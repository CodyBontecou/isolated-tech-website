## Implementation
```typescript
// lib/referral.ts
import { nanoid } from "./db";

export function generateReferralCode(userName?: string): string {
  const base = userName 
    ? userName.toUpperCase().replace(/[^A-Z]/g, "").slice(0, 6)
    : "REF";
  const suffix = nanoid(4).toUpperCase();
  return `${base}${suffix}`;
}

export async function getOrCreateReferralCode(userId: string, env: Env) {
  // Check if user has existing code
  let code = await env.DB.prepare(
    "SELECT * FROM referral_codes WHERE user_id = ? AND is_active = 1"
  ).bind(userId).first();
  
  if (code) return code;
  
  // Check if user has made a purchase (required to get code)
  const purchase = await env.DB.prepare(
    "SELECT id FROM purchases WHERE user_id = ? AND status = 'completed' LIMIT 1"
  ).bind(userId).first();
  
  if (!purchase) {
    throw new Error("Must make a purchase before getting a referral code");
  }
  
  // Get user info for code generation
  const user = await env.DB.prepare(
    "SELECT name FROM user WHERE id = ?"
  ).bind(userId).first();
  
  // Create new code
  const newCode = {
    id: nanoid(),
    user_id: userId,
    code: generateReferralCode(user?.name),
    discount_percent: 20,
    reward_cents: 500,
  };
  
  await env.DB.prepare(`
    INSERT INTO referral_codes (id, user_id, code, discount_percent, reward_cents)
    VALUES (?, ?, ?, ?, ?)
  `).bind(newCode.id, newCode.user_id, newCode.code, newCode.discount_percent, newCode.reward_cents).run();
  
  return newCode;
}

export async function validateReferralCode(code: string, userId: string, env: Env) {
  const referral = await env.DB.prepare(`
    SELECT rc.*, u.name as owner_name
    FROM referral_codes rc
    JOIN user u ON rc.user_id = u.id
    WHERE rc.code = ? AND rc.is_active = 1
  `).bind(code.toUpperCase()).first();
  
  if (!referral) return null;
  
  // Can't use own code
  if (referral.user_id === userId) return null;
  
  // Check max uses
  if (referral.max_uses && referral.times_used >= referral.max_uses) return null;
  
  return referral;
}
```

## API Route
```typescript
// app/api/referral/code/route.ts
export async function GET() {
  const env = getEnv();
  const user = await getCurrentUser(env);
  
  if (!user) {
    return Response.json({ error: "Auth required" }, { status: 401 });
  }
  
  try {
    const code = await getOrCreateReferralCode(user.id, env);
    return Response.json({ code });
  } catch (error) {
    return Response.json({ error: error.message }, { status: 400 });
  }
}
```
