## Tasks
1. Create `app/stats/page.tsx`
2. Design stat card grid
3. Add "last updated" timestamp
4. Match site aesthetic

## Implementation
```typescript
// app/stats/page.tsx
import { getPublicStats } from "@/lib/stats";
import { SiteNav } from "@/components/site-nav";
import { SiteFooter } from "@/components/site-footer";

export const metadata = {
  title: "Stats — ISOLATED.TECH",
  description: "Transparency report showing our aggregate metrics.",
};

function StatCard({ value, label, icon }: { value: string | number; label: string; icon?: string }) {
  return (
    <div className="stat-card">
      {icon && <span className="stat-card__icon">{icon}</span>}
      <span className="stat-card__value">{value}</span>
      <span className="stat-card__label">{label}</span>
    </div>
  );
}

export default async function StatsPage() {
  const stats = await getPublicStats();
  
  return (
    <>
      <SiteNav user={null} activePage="stats" />
      
      <main className="stats-page">
        <header className="stats-header">
          <h1 className="stats-header__title">STATS</h1>
          <p className="stats-header__subtitle">
            Transparency report · Updated hourly
          </p>
        </header>
        
        <div className="stats-grid">
          <StatCard value={stats.apps} label="APPS" icon="📦" />
          <StatCard value={stats.downloads.toLocaleString()} label="DOWNLOADS" icon="⬇️" />
          <StatCard value={stats.reviews} label="REVIEWS" icon="⭐" />
          <StatCard 
            value={stats.avgRating ? stats.avgRating.toFixed(1) : "—"} 
            label="AVG RATING" 
            icon="★" 
          />
          <StatCard value={stats.countries} label="COUNTRIES" icon="🌍" />
          <StatCard value={stats.feedback} label="FEEDBACK" icon="💬" />
        </div>
        
        <p className="stats-updated">
          Last updated: {new Date(stats.updatedAt).toLocaleString()}
        </p>
      </main>
      
      <SiteFooter />
    </>
  );
}
```
