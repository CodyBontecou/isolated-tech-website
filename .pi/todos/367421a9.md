## Implementation
```typescript
// lib/stats.ts
import { getEnv, type Env } from "./env";

export interface PublicStats {
  apps: number;
  downloads: number;
  reviews: number;
  avgRating: number | null;
  countries: number;
  feedback: number;
  updatedAt: string;
}

export async function getPublicStats(env?: Env): Promise<PublicStats> {
  const e = env || getEnv();
  
  // Check KV cache first
  const cached = await e.KV?.get("public-stats", "json") as PublicStats | null;
  if (cached) return cached;
  
  // Compute fresh stats
  const [apps, downloads, reviewData, countries, feedback] = await e.DB.batch([
    e.DB.prepare("SELECT COUNT(*) as count FROM apps WHERE is_published = 1"),
    e.DB.prepare("SELECT COUNT(*) as count FROM downloads"),
    e.DB.prepare(`
      SELECT COUNT(*) as count, AVG(rating) as avg_rating FROM (
        SELECT rating FROM reviews WHERE is_approved = 1
        UNION ALL SELECT rating FROM app_store_reviews
      )
    `),
    e.DB.prepare("SELECT COUNT(DISTINCT country) as count FROM downloads WHERE country IS NOT NULL"),
    e.DB.prepare("SELECT COUNT(*) as count FROM feature_requests WHERE status != 'closed'"),
  ]);
  
  const stats: PublicStats = {
    apps: apps.results?.[0]?.count || 0,
    downloads: downloads.results?.[0]?.count || 0,
    reviews: reviewData.results?.[0]?.count || 0,
    avgRating: reviewData.results?.[0]?.avg_rating || null,
    countries: countries.results?.[0]?.count || 0,
    feedback: feedback.results?.[0]?.count || 0,
    updatedAt: new Date().toISOString(),
  };
  
  // Cache for 1 hour
  await e.KV?.put("public-stats", JSON.stringify(stats), { expirationTtl: 3600 });
  
  return stats;
}
```
