Parent: Epic: Dynamic OpenGraph Images

## Tasks
1. Create `app/api/og/[slug]/route.ts`
2. Fetch app data from database
3. Generate OG image with Satori
4. Convert to PNG with Resvg
5. Cache response appropriately

## Implementation
```typescript
// app/api/og/[slug]/route.ts
import satori from "satori";
import { Resvg } from "@resvg/resvg-wasm";
import { getEnv } from "@/lib/cloudflare-context";
import { loadFonts } from "@/lib/og-fonts";
import { OGImage } from "@/lib/og-image";

export async function GET(
  request: Request,
  { params }: { params: Promise<{ slug: string }> }
) {
  const { slug } = await params;
  const env = getEnv();
  
  const app = await env.DB.prepare(
    "SELECT name, tagline, icon_url FROM apps WHERE slug = ? AND is_published = 1"
  ).bind(slug).first();
  
  if (!app) {
    return new Response("Not found", { status: 404 });
  }
  
  const fonts = await loadFonts();
  
  const svg = await satori(
    OGImage({ name: app.name, tagline: app.tagline, iconUrl: app.icon_url }),
    { width: 1200, height: 630, fonts }
  );
  
  const resvg = new Resvg(svg);
  const png = resvg.render().asPng();
  
  return new Response(png, {
    headers: {
      "Content-Type": "image/png",
      "Cache-Control": "public, max-age=86400, s-maxage=604800",
    },
  });
}
```
