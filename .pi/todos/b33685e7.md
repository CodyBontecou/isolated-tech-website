// app/dashboard/referrals/page.tsx

async function getReferralData(userId: string, env: Env) {
  const [code, referrals, user] = await Promise.all([
    env.DB.prepare(
      "SELECT * FROM referral_codes WHERE user_id = ? AND is_active = 1"
    ).bind(userId).first(),
    env.DB.prepare(`
      SELECT r.*, u.email as referred_email, p.amount_cents
      FROM referrals r
      JOIN user u ON r.referred_user_id = u.id
      JOIN purchases p ON r.purchase_id = p.id
      WHERE r.referrer_user_id = ?
      ORDER BY r.created_at DESC
    `).bind(userId).all(),
    env.DB.prepare(
      "SELECT referral_credit_cents FROM user WHERE id = ?"
    ).bind(userId).first(),
  ]);
  
  return {
    code,
    referrals: referrals.results,
    creditCents: user?.referral_credit_cents || 0,
  };
}
